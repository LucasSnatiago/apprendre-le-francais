\section{Écrire le bonne terminaison}

\begin{enumerate}
    \item Je parl\und{e}.
    \item Tu chant\und{es}.
    \item Elle dans\und{e}.
    \item Nous étudi\und{ons}.
    \item Vous jou\und{ez}.
    \item Ils écout\und{ent}.
\end{enumerate}

\section{Écrire au présent}

\begin{enumerate}
    \item Pierre et Catherine \und{parlent} et \und{regardent} des photos.
    \item Tu \und{chantes} une jolie chanson et je \und{danse}.
    \item Nicolas \und{aime} le chocolat et \und{déteste} les carottes.
    \item Nous \und{étudions} le français: nous \und{travaillons} beaucoup.
    \item La petite fille jou\und{e} dans le jardin.
    \item Vous écout\und{ez} un disque et elles discut\und{ent}.
\end{enumerate}

\section{Écrire au présent:}

\begin{enumerate}
    \item Vous \und{appelez} le chien, j'\und{appelle} les enfants.
    \item Elle \und{jette} des papiers, vous \und{jetez} aussi.
    \item Il neige: je \und{gelle}, vous \und{gelez} aussi.
    \item Ils \und{achettent} des diques, nous \und{achetons} des livres.
    \item Tu \und{pelles} une poire, nous \und{pelons} des pommes.
\end{enumerate}
