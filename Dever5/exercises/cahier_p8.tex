\section{C'est \txb{ou} il/elle est?}

\begin{enumerate}
    \item \und{C'est} elle.
    \item \und{Elle est} grande.
    \item \und{C'est} une fille sympathique.
    \item \und{C'est} dans sa chambre.
    \item Lui, \und{Il est} agent de voyages.
    \item \und{Elle est} sympathique.
    \item Elle, \und{C'est} une femme célèbre.
    \item Lui, \und{C'est} un garçon sérieux. 
\end{enumerate}

\section{C'est \txb{ou} il/elle est?}

\begin{enumerate}
    \item La jeune femme, là-bas, \und{Elle est} italienne?
    \item Oui, \und{C'est} une italienne.
    \item \und{C'est} une étudiante?
    \item Non, \und{Elle est} secrétaire.
    \item Et le jeune homme avec elle, \und{C'est} un ami?
    \item Oui, \und{Il est} sympathique.
    \item Et \und{C'est} un beau couple.
\end{enumerate}
