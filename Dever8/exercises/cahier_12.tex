\section{Adjectifs possessifs}

\begin{enumerate}
    \item Voici \und{ma} femme et \und{ma} fille.
    \item Je vous présente \und{mon} pêre et \und{ma} mère.
    \item C'est \und{votre} chambre? $-$ Oui, C'est \und{ma} chambre.
    \item C'est \und{ta} amie? $-$ Non, c'est \und{mon} cuisinière.
    \item Tu as \und{votre} adresse? $-$ Non, mais j'ai \und{mon} numéro de téléphone.
\end{enumerate}
